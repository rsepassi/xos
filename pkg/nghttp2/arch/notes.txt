# To get config.h

ARCH_ISA=aarch64
ARCH_OS=windows
lib=gnu

ARCH=$ARCH_ISA-$ARCH_OS-$lib
ARCH_HOST=aarch64-darwin

CC="zig cc --target=$ARCH" \
./configure \
  --prefix=$PWD/build \
  --host=$ARCH_ISA-$ARCH_OS \
  --build=$ARCH_HOST \
  --disable-shared \
  --disable-dependency-tracking \
  --enable-lib-only

mkdir -p arch
cp config.h arch/config-$ARCH_OS.h
